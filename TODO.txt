
<!--  Bloque normal -->
span
    <!-- Se traduce a -->
    <span></span>


<!--  Bloque con clase -->
span(red label)
    <!-- Se traduce a -->
    <span class="red label"></span>


<!--  Bloque con attributo -->
span[attr="attrValue", value="name"]
    <!-- Se traduce a -->
    <span attr="attrValue" value="name"></span>


<!--  Bloque con texto -->
span "Hola"
    <!-- Se traduce a -->
    <span>Hola</span>
